GIT

// wypisuwanie branch-y
git branch

// dodawanie nowego brancha
git branch nazwa-brancha

//przechodzenia na brancha
git checkout nazwa-brancha

//sprawdzanie  zmian
git status 

//dodanie zmian
git add .

//dodanie commita
git commit -am 'co zrobilismy'

// dodanie brancha do mastera
git push origin nazwa-brancha

// usuwanie brancha
git branch -D nazwa-brancha

//aktualizowanie zmian na mastera
git pull
